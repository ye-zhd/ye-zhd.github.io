ansible学习使用

发表于2021-11-27|更新于2021-11-27|Linux
|阅读量:

ansible

crontab 配置ansible 注意事项

#需要配置 PYTHONPATH
# import os 
# print os.sys.path 查看pythonpath
export PYTHONPATH=/usr/lib/python2.6/site-packages
ansible-playbook /tmp/test.yaml -s >>/tmp/ansible-cron.log

ansible cron

#ansible 添加 定时任务
ansible 10.252.10.101 -m cron -a "minute=*/10 hour=* day=* month=*   weekday=* name='test' job=\"sh /data/etl/mv_data.sh\"" -S -R root

#ansible 注释任务
ansible 10.252.10.101 -m cron -a "name='test' job=\"sh /data/etl/mv_data.sh\" disabled=yes" -S -R root
#ansible 删除任务
ansible 10.252.10.101 -m cron -a "name=test state=absent"

#####sed 查找替换 定时任务时间周期
sed -i '/mv_data/s/30/5/g' /var/spool/cron/root

ansible-cmdb

wget https://github.com/fboender/ansible-cmdb/releases/download/1.30/ansible-cmdb-1.30.tar.gz
tar -xvf ansible-cmdb-1.30.tar.gz -C /usr/share/ansible/plugins
cd /usr/share/ansible/plugins/ansible-cmdb-1.30/ && make install

#生成所有主机的facts信息
ansible all -m setup --tree /usr/local/out
#生成的facts信息生成web页面
ansible-cmdb /usr/local/out/ > overview.html
#以资产列表形式统计出ansible主机信息
ansible-cmdb -t txt_table --columns name,os,ip,mem,cpus /usr/local/out/

#more命令
more */*/*.txt |grep '602.E602E22'
#批量改名字
rename .ok .xml *.ok

ansible playbook

[root@ansible PlayBook]# ansible-playbook -h
#ansible-playbook常用选项：
--check  or -C    #只检测可能会发生的改变，但不真正执行操作
--list-hosts      #列出运行任务的主机
--list-tags       #列出playbook文件中定义所有的tags
--list-tasks      #列出playbook文件中定义的所以任务集
--limit           #主机列表 只针对主机列表中的某个主机或者某个组执行
-f                #指定并发数，默认为5个
-t                #指定tags运行，运行某一个或者多个tags。（前提playbook中有定义tags）
-v                #显示过程  -vv  -vvv更详细

root权限执行：  ansible-playbook  -S -R root test.yaml

例子

批量拷贝文件到本地

##批量拷贝文件到本地
- hosts: cxchuzhi:gmqchuzhi
  tasks:
    - name: result_shazhupan
      find:
        paths: /tmp/
        patterns: "*xml"
        recurse: no
      register: file_2_fetch

    - name: data_shazhupan
      fetch:
        src: "{{ item.path }}"
        dest: /tmp/linux_check/
        flat: yes
      with_items: "{{ file_2_fetch.files }}"

添加修改用户yaml

##添加用户并sudo
- hosts: test
  vars:
    user: hdfs
    password: 'eversec123098'
  remote_user: root
  tasks:
  - name: Add user {{ user }}
    user: name={{user}} comment="ambari user" password='eversec123098'
  - name: Config /etc/sudoers
    lineinfile: dest=/etc/sudoers state=present  line='{{item}}'
    with_items:
      - "{{ user}} ALL=(ALL) NOPASSWD: ALL"
      - "Defaults: {{user}}  !requiretty"
- hosts: px
  gather_facts: false
  tasks:
  - name: Change password
    user: name={{ name }}  password={{ chpass | password_hash('sha512') }}  update_password=always
  - name: Config /etc/sudoers
    lineinfile: dest=/etc/sudoers state=present  line='{{item}}'
    with_items:
      - "{{ name }} ALL=(ALL) NOPASSWD: ALL"
      - "Defaults: {{name}}  !requiretty

服务启动：

---                       #固定格式
- hosts: 10.20.3.107      #定义需要执行主机
  remote_user: root       #远程用户
  vars:                   #定义变量
    http_port: 8088       #变量
  tasks:                             #定义一个任务的开始
    - name: start crond
      service: name=crond state=started
